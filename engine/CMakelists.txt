cmake_minimum_required(VERSION 3.12)
project(PhysicsEngine VERSION 1.0.0)

file(GLOB_RECURSE INCLUDES "include/*.h*")
file(GLOB_RECURSE SOURCES "src/*.cpp*")

add_library(
	engine STATIC
	${INCLUDES}
	${SOURCES}
)

if ( MSVC )
	target_compile_options(engine PRIVATE "/W4")
    target_compile_options(engine PRIVATE "-wd4100")
    target_compile_options(engine PRIVATE "-wd4996")
    target_compile_options(engine PRIVATE "-wd4211")
    target_compile_options(engine PRIVATE "/EHsc")
    target_compile_options(engine PRIVATE "$<$<CONFIG:DEBUG>:/Od;/MDd;-Zi>")
    target_compile_options(engine PRIVATE "$<$<CONFIG:RELEASE>:/O2;/MD>")
endif()

target_include_directories(engine PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/include/glew-2.1.0/")
target_include_directories(engine PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/include/freetype/")

target_compile_definitions(engine PUBLIC MY_DEBUG_DEFINE=0)
